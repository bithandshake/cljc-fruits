(defn not-starts-with!
  [n x]
  (let [n (str n)
        x (re-pattern x)]
       (or (if-let [first-match (match/re-first n x)]
                   (if (clojure.string/starts-with? n first-match)
                       (subs n (-> first-match count))))
           (-> n))))

(defn first-dex-of
  [n x]
  (let [n (str n)
        x (re-pattern x)]
       (if-let [match (re-find x n)]
               (cond (vector? match) (string/first-dex-of n (first match))
                     (string? match) (string/first-dex-of n        match)))))

(defn nth-dex-of
  [n x th]
  (let [n (str n)
        x (re-pattern x)]
       (if (nat-int? th)
           (loop [cursor 0 skip 0]
                 (if-let [first-dex (-> n (string/keep-range cursor)
                                          (first-dex-of x))]
                         (if (= skip th)
                             (+ cursor first-dex)
                             (recur (+ first-dex cursor 1)
                                    (inc skip))))))))

(defn re-from
  [n x cursor]
  (let [n       (str n)
        x       (re-pattern x)
        cursor  (seqable/normalize-cursor n cursor {:adjust? true :mirror? true})
        matches (re-seq x n)]
       (letfn [(f0 [result match]
                   (or (and (seqable/cursor-in-bounds? n (+ cursor (count match)))
                            (=  match (subs n cursor (+ cursor (count match))))
                            (-> match count (> (count result)))
                            (-> match))
                       (-> result)))]
              (reduce f0 nil matches))))

(defn remove-first-match
  [n x]
  (let [n (str n)
        x (re-pattern x)]
       (if-let [first-match (match/re-first n x)]
               (let [dex (clojure.string/index-of n first-match)]
                    (str (subs n 0 dex)
                         (subs n (+ dex (count first-match)))))
               (-> n))))

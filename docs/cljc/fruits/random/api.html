(defn generate-integer
  ([]
   (generate-integer 4))

  ([digits]
   (if (integer? digits)
       (int (min (*   (math/power 10 (-> digits dec)) (-> 9 rand inc))
                 (dec (math/power 10 (-> digits))))))))

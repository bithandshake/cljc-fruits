(defn unkeywordized-value?
  [n]
  (and (-> n string?)
       (-> n count (> 2))
       (= config/KEYWORD-PREFIX (str (nth n 0)))
       (= ":"                   (str (nth n 1)))))

(defn remove-blank-values
  [n]
  (letfn [(f [x] (vector/contains-item? [{} [] () nil ""] x))]
         (let [result (map/remove-values-by n f {:recur? true})]
              (if (-> result (= n))
                  (-> result)
                  (-> result remove-blank-values)))))

integer->lowercase
(defn integer->lowercase
  [n]
  (let [base-char (int \a)]
       (loop [num n result ""]
             (if (-> num zero?)
                 (-> result)
                 (let [char (char (+ base-char (mod (dec num) 26)))]
                      (recur (quot (dec num) 26) (str char result)))))))

(defn normalize-cursor
  ([n cursor]
   (normalize-cursor n cursor {}))

  ([n cursor {:keys [adjust? mirror?]}]
   (let [n      (mixed/to-seqable n)
         cursor (mixed/to-integer cursor)]
        (letfn [(f0 [%] (-> n count inc (+ %)))
                (f1 [%] (-> n count (>= %)))]
               (if (-> cursor neg-int?)
                   (cond (-> mirror?)   (-> n (normalize-cursor (f0 cursor) {:adjust? adjust?})) 
                         (-> adjust?)   (-> 0))                                                  
                   (cond (-> cursor f1) (-> cursor)                                              
                         (-> adjust?)   (-> n count)))))))

  [n path]
  (let [path (mixed/to-vector path)]
       (letfn [(f0 [result k]
                   (if (fn? k)
                       (conj result (-> n (get-in result) k))
                       (conj result (-> k))))]
              (reduce f0 [] path))))

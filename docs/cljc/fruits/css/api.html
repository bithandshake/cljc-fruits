(defn unparse
  [n]
  (-> n (map/to-vector   (fn [k v] [k v]))
        (vector/->>items (fn [x] (keyword/to-string x)))
        (vector/->items  (fn [x] (string/join x ": ")))
        (vector/suffix-items "; ")
        (string/join)
        (string/trim)))

(defn parse
  [n]
  (-> n (string/split    #";")
        (vector/->items  (fn [x] (string/split x #":")))
        (vector/->>items (fn [x] (string/trim  x)))
        (vector/to-map   (fn [_ [k v]] [(keyword k) v]))))

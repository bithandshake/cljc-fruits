(defn to-string
  [n]
  (if (keyword? n)
      (if-let [namespace (namespace n)]
              (str namespace "/" (name n))
              (name n))
      (str n)))

(defn prepend
  ([n x]
   (if-let [namespace (namespace n)]
           (keyword namespace (str (name x) (name n)))
           (keyword (str (name x) (name n)))))

(defn join
  ([n]
   (join n nil))

  ([n delimiter]
   (letfn [(f0 [result x]
               (if (keyword? x) (str result (if result delimiter) (name x))
                                (str result (if result delimiter) x)))]
